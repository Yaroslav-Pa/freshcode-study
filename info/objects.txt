  насправді усі об'єкти це посилання на комірку пам'яті тому: 

  const obj1 = {}
  const obj2 = {}

  return obj1 === obj1 //true
  return obj1 === obj2 // fasle
  тому що obj1, наприклад, буде посланням на клітинку 5000, але obj2, знову наприклад, посилання на 6500

  !!! Але якщо зробити: 
  const obj3 = obj1;

  return obj1 === obj3 // true
  !!! тому що тепер obj3 дорівнює посиланню на комірку 5000, як й obj1, тож якщо змінити obj1 або obj3 будуть змінені вони вдвох (бо зміни будуть записані у спільну комірку)


  також зробивши так: 
  function testFunc (someObj) {
  someObj.someValue = 500;

  return someObj;
  }

  const lastObject = testFunc(obj1);

  цей об'єкт все ще посилається на одну й туж комірку тож будуть змінені вони вдвох
